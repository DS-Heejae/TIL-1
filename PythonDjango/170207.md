# 파이썬 기본기 익히기 다섯번째 

## 1. 클래스 


```py 
# Shape class
class Shape:
	type = None
	solidColor = None
	strokeColor = None
	x = 0
	y = 0
	width = 0
	height = 0

	def __init__(self, type, solidcolor, strokecolor, x, y):
		self.type = type
		self.solidColor = solidcolor
		self.strokeColor = strokecolor
		self.x = x
		self.y = y

	def setSize(self, width, height):
		self.width = width
		self.height = height
```
